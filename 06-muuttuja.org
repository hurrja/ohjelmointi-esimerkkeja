  #+BEGIN_SRC processing :exports code
    size (600, 600);

    final float KX = 400; // keskipisteen x-koordinaatti

    float r; // ympyrän säde

    if (KX < width / 2)
      r = KX;
    else
      r = width - KX;

    float d = 2 * r; // halkaisija
    ellipse (KX, KX, d, d);
